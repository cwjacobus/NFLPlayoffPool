CREATE TABLE User
(
UserId int PRIMARY KEY AUTO_INCREMENT,
UserName varchar(255) NOT NULL,
LastName varchar(255),
FirstName varchar(255),
Email varchar(255),
Year int(5),
admin boolean,
PoolId int(5)
);

CREATE TABLE NFLPlayoffsGame
(
GameIndex int PRIMARY KEY AUTO_INCREMENT,
Description varchar(255) NOT NULL,
Winner varchar(25),
Loser varchar(25),
PointsValue int,
Completed boolean,
Year int(5),
Home int(5),
Visitor int(5),
Conference varchar(5);
HomeScore int(3);
VisScore int(3);
HomeFav boolean;
HomeSeed int(3);
VisSeed int(3);
DateTime timestamp,
);

CREATE TABLE Pick
(
PickId int PRIMARY KEY AUTO_INCREMENT,
UserId int,
GameId int,
Winner varchar(25),
PoolId int(5),
CreatedTime timestamp
);

CREATE TABLE Pool
(
PoolId int(5) PRIMARY KEY AUTO_INCREMENT,
PoolName varchar(255) NOT NULL,
Year int(5) NOT NULL,
FirstGameDateTime timestamp,
PointsRd1 int(11) NOT NULL DEFAULT 3,
PointsRd2 int(11) NOT NULL DEFAULT 5,
PointsChamp int(11) NOT NULL DEFAULT 10,
PointsSB int(11) NOT NULL DEFAULT 20,
CONSTRAINT P_CONST1 UNIQUE (PoolName, Year)
);

CREATE TABLE NFLTeam
(
NFLTeamId int(5) PRIMARY KEY,
LongName varchar(50) NOT NULL,
ShortName varchar(5) NOT NULL,
Conference varchar(5) NOT NULL
);

CREATE TABLE NFLPlayoffsTeam
(
NFLPlayoffsTeamId int(5) PRIMARY KEY AUTO_INCREMENT,
NFLTeamId int(5) NOT NULL,
Seed int(5) NOT NULL,
Year int(5) NOT NULL
);

//Get Winners/Losers using scores
SELECT
  CASE 
    WHEN g.homescore > g.visscore THEN t1.shortname
    WHEN g.visscore > g.homescore THEN t2.shortname
    ELSE 'Tie'
  END as Winner,
  CASE 
    WHEN g.homescore < g.visscore THEN t1.shortname
    WHEN g.visscore < g.homescore THEN t2.shortname
    ELSE 'Tie'
  END as Loser
FROM nflplayoffsgame g, nflteam t1, nflteam t2 where g.home = t1.nflteamid and g.visitor = t2.nflteamid and g.year = 20 and g.completed=1;
